import { useState, useEffect } from "react";

export const Views = () => {
  const [data, setData] = useState(null);
  useEffect(() => {
    fetch("api/views", {
      method: "POST",
      mode: "same-origin",
      cache: "no-cache",
      headers: {
        "Content-Type": "application/json",
      },
    })
      .then((response) => response.json())
      .then((json) => setData(json))
      .catch((error) => console.error(error));
  }, []);
  if (!data) {
    return "Fetching views...";
  }
  return data;
};
